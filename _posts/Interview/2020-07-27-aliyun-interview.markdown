memcpy的实现，源地址与目的地址有重叠
```
void * mymemncpy(void* src, void* dst, int len)
{
    if (NULL == dst || NULL == src)
    {
        return NULL;
    }
  
    char* dstTmp = char*(dst);
    char* srcTmp = char*(src);
    int i;
  
    //重叠部分处理
    if (dstTmp > srcTmp && dstTmp < srcTmp + len)
    {
        for (i = len -1; i != -1; --i)
        {
            dstTmp[i] = srcTmp[i];
        }      
    }
    else
    {
        for (i = 0, i < len; i++)
        {
            dstTmp[i] = srcTmp[i]; 
        }
    }
  
    return dst;
}
```
